        // Function to fetch and parse all notes for graph

        async function buildGraph() {

            const nodes = new vis.DataSet();

            const edges = new vis.DataSet();

            const noteMap = new Map(); // filename to id


            let id = 1;

            for (const file of noteFiles) {

                const response = await fetch(`notes/${file}`);

                if (response.ok) {

                    const content = await response.text();

                    nodes.add({ id: id, label: file.replace('.md', '') });

                    noteMap.set(file, id);


                    // Extract links: [[link]]

                    const links = content.match(/\[\[([^\]]+)\]\]/g) || [];

                    for (const link of links) {

                        const target = link.slice(2, -2) + '.md'; // Assume link is to note filename

                        if (noteFiles.includes(target)) {

                            const targetId = noteMap.get(target);

                            if (targetId) {

                                edges.add({ from: id, to: targetId });

                            }

                        }

                    }

                    id++;

                }

            }


            // Render graph

            const container = document.getElementById('graph-container');

            const data = { nodes, edges };

            const options = {

                layout: { hierarchical: false },

                physics: { enabled: true },

                nodes: { shape: 'dot', size: 10 },

                edges: { arrows: 'to' }

            };

            new vis.Network(container, data, options);

        }


        // Function to list notes

        function listNotes() {

            const ul = document.getElementById('notes-list');

            noteFiles.forEach(file => {

                const li = document.createElement('li');

                const a = document.createElement('a');

                a.href = '#';

                a.textContent = file.replace('.md', '');

                a.onclick = () => loadNote(file);

                li.appendChild(a);

                ul.appendChild(li);

            });

        }


        // Function to load and render a note

        async function loadNote(file) {

            const response = await fetch(`notes/${file}`);

            if (response.ok) {

                const markdown = await response.text();

                const html = marked.parse(markdown);

                document.getElementById('note-content').innerHTML = html;

            }

        }


        // Initialize

        buildGraph();

        listNotes();
